1. Define User (models.ts) 
import mongoose from "mongoose";

const userSchema = new mongoose.Schema({
  email: { type: String, required: true, unique: true },
  passwordHash: { type: String },
  firstName: { type: String },
  nin: { type: String } // Native extended field!
});

export const User = mongoose.model("User", userSchema);
2. Create Adapter (adapter.ts) 
import type { AuthAdapter } from "kroxt/adapter";
import { User } from "./models.js";

// This is where you connect the `auth-lib` to MongoDB.
export const mongoAdapter: AuthAdapter<any> = {
  createUser: async (data: any) => {
    const newUser = await User.create(data);
    return {
      id: newUser._id.toString(),
      email: newUser.email,
      passwordHash: newUser.passwordHash,
      role: newUser.role,
      ...newUser.toObject(),
    };
  },
  
  findUserByEmail: async (email: string) => {
    const user = await User.findOne({ email });
    if (!user) return null;
    return { id: user._id.toString(), ...user.toObject() };
  },

  findUserById: async (id: string) => {
    const user = await User.findById(id);
    if (!user) return null;
    return { id: user._id.toString(), ...user.toObject() };
  },

  linkOAuthAccount: async (userId: string, provider: string, providerId: string) => {
    console.log(`[MongoAdapter] Linked OAuth account ${providerId} from ${provider} to ${userId}`);
  }
};
3. Initialize Engine (index.ts) 
import { createAuth } from "kroxt";
import { mongoAdapter } from "./adapter.js";

const auth = createAuth({
    adapter: mongoAdapter,
    secret: "my-super-secret-mongo-key-123",
    session: { expires: "7d" }
});

// Ready! auth.signup(), auth.loginWithPassword() 🚀

The Core Interface
export interface BaseUser {
  id: string;
  email: string;
  passwordHash?: string;
  role?: string;
}

// Supports any extended fields (nin, bvn, role, etc.)
export type User<TExtended = Record<string, any>> = BaseUser & TExtended;

export interface AuthAdapter<TUser = User> {
  createUser: (data: any) => Promise<TUser>;
  findUserByEmail: (email: string) => Promise<TUser | null>;
  findUserById: (id: string) => Promise<TUser | null>;
  linkOAuthAccount: (uId: string, p: string, pId: string) => Promise<void>;
}
